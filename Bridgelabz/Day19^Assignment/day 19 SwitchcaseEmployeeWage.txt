    public static final int isPartTime=0;

	public static void main(String[] args) {

      int emphrs=20;
      int empwage=0;
      double empCheck=Math.floor(Math.random()*10)%3;
    switch ((int)empCheck)

               emphrs=8;

                    emphrs=4;


	default : System.out.println("fulltimeemployee wage");

    }
}
